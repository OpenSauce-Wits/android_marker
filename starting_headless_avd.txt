$SDK_MANAGER ~ dir to sdkmanager binary i.e $ANDROID_SDK_ROOT/tools/bin/sdkmanager
$AVD_MANAGER ~ i.e $ANDROID_SDK_ROOT/tools/bin/avdmanager
$EMULATOR i.e $ANDROID_SDK_ROOT/emulator/emulator

#update emulator
$SDK_MANAGER --verbose --channel=0 emulator

#update system-image packages
$SDK_MANAGER --verbose "system-images;android-29;google_apis;x86"

#create avd
$AVD_MANAGER --verbose create avd --force --name emulator --device pixel --package "system-images;android-29;google_apis;x86" --tag google_apis --abi x86 --sdcard 512M

#start avd
$EMULATOR @emulator -show-kernel -no-audio -no-window -no-boot-anim -netdelay none -no-snapshot -wipe-data -gpu swiftshader_indirect -camera-back none -camera-front none

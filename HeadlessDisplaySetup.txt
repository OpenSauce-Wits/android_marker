Adding virtual display to my server
Installing x11-xserver-utils
mkdir /usr/share/X11/xorg.conf.d/


Section "Device"
    Identifier "intelgpu0"
    Driver "intel"
    Option "VirtualHeads" "1"
EndSection

>> /usr/share/X11/xorg.conf.d/20-intel.conf
logout then login

install xserver-xorg-video-dummy ~ dummy dispay driver which isn't gpu accelerated

Section "Device"
    Identifier  "Configured Video Device"
    Driver      "dummy"
EndSection

Section "Monitor"
    Identifier  "Configured Monitor"
    HorizSync 31.5-48.5
    VertRefresh 50-70
EndSection

Section "Screen"
    Identifier  "Default Screen"
    Monitor     "Configured Monitor"
    Device      "Configured Video Device"
    DefaultDepth 24
    SubSection "Display"
    	Depth 24
    	Modes "1360x768"
    EndSubSection
EndSection

>> /etc/X11/xorg.conf

rm /usr/share/X11/xorg.conf.d/20-intel.conf

X :<DISPLAY_NO> -config /etc/X11/xorg.conf

GRUB_GFXPAYLOAD_LINUX=1024x768
>> /etc/default/grub to change default resolution of ubuntu server
update-grub
reboot
